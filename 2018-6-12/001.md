======所以我用了別的方法=======

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main()
{
   
    ifstream input("c:///scores33.txt");   //開啟固定的檔案
    
    ofstream output;
 
    output.open("c:///scores31.txt");      //新建一個檔案
    
    string firstName;
    char mi;
    string lastName;
    int score;
  
    input >> firstName >> mi >> lastName >> score;
    cout << firstName << " " << mi << " " << lastName << " "
        << score+4 << endl;
    output << firstName << " " << mi << " " << lastName << " "
        << score+4 << endl;

    input >> firstName >> mi >> lastName >> score;
    cout << firstName << " " << mi << " " << lastName << " "
        << score+4 << endl;
    output << firstName << " " << mi << " " << lastName << " "
        << score+4 << endl;
    
    input >> firstName >> mi >> lastName >> score;
    cout << firstName << " " << mi << " " << lastName << "  "
        << score+4 << endl; 
    output << firstName << " " << mi << " " << lastName << " "
        << score+4 << endl;
    
    output.close();
  
    input.close();

    cout << "Done" << endl;

    return 0;
}
